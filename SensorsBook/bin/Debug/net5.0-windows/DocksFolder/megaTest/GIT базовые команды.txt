//see status = on what branch I'm currently and what happens with changes,
//is there need to add or commite.
git status

//adding to git to control changes
git add <file name>

//commiting = saving current "snapshot" of files
git commit -m "messege following changes I'm commiting"

//push = sending changes to remout git server
git push

//see what branch is it
git branch

//change branch
git checkout [branch name]

//see difference between previous commit and current version
git diff

//see log = history of commits
git log

//see all files
git ls-files

//remove from repository
git rm <fileName1> <fileName2> ... <fileNameN>

git rm -r <directory>
